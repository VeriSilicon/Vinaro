85/*
86 * NOTES:
87 * In general, configuring the following items, then you can using DMA
88 * correctly. if you don't know what's means of other parameters,
89 * please leave them alone.
90 */
91typedef struct DmacChannelCfg {
92 /*
93 * chn_id will be assigned as the same with the handshake id
94 * please refer to the DmacHandShakeId
95 * NOTE: channel number starts from 0,1...
96 */
97 uint8_t chn_id;
98
99 /*
100 * which priority level(0~7) do you want to assign to channel.
101 * A priority of 7 is the highest priority, and 0 is the lowest.
102 */
103 uint8_t prio;
104
105 /* handshake signal id. Only used when peripherals are non-memory */
106 uint8_t src_hs_id;
107 uint8_t dst_hs_id;
108
109 /* dma_intr: 0 disable dmac interrupt, 1 enable dmac interrupt. */
110 uint8_t dma_intr;
111
112 /*
113 * the direction of this transfer.
114 * DMA_MEM_TO_MEM: Memory to Memory.
115 * DMA_MEM_TO_DEV: Memory to Device.
116 * DMA_DEV_TO_MEM: Device to Memory.
117 * DMA_DEV_TO_DEV: Device to Device.
118 */
119 uint8_t dir;
120
121 /*
122 * The change way of source or destination address, after every source
123 * or destination transfer.
124 * 00 = Increment; 01 = Decrement; 1x = No change
125 */
126 uint8_t sinc;
127 uint8_t dinc;
128
129 /*
130 * Source or destination peripherals transfer width.
131 * Generally, we assign separately src_tr_width and dst_tr_width with
132 * the width of peripherals FIFO.
133 */
134 uint8_t src_tr_width;
135 uint8_t dst_tr_width;
136
137 /*
138 * Flow Control Mode
139 * Determines when source transaction requests are serviced when
140 * the Destination Peripheral is the flow controller.
141 * 0x0 (FCMODE_0): Data pre-fetching is enabled(default)
142 * 0x1 (FCMODE_1): Data pre-fetching is disabled
143 */
144 uint8_t fc_mode;
145
146 /*
147 * how many bytes you want to transfer.
148 */
149 uint32_t len;
150
151 /*
152 * source or destination addresses.
153 */
154 uint32_t src_addr;
155 uint32_t dst_addr;
156
157 /*
158 * Source or destination Burst Transaction Length, instead of AHB burst
159 * transfer length.
160 * it's meaning that Number of data items, each of width src_tr_width
161 * or dst_tr_width (NOT BYTE NUMBER), to be read/write from/to the
162 * source/destination every time a source burst transaction.
163 */
164 uint32_t src_msize;
165 uint32_t dst_msize;
166
167 /*
168 * Configure items of linked list multi-blocks transfer.
169 * you can separately enable blocking chaining either source or
170 * destination side by assigning with llp_src_en and llp_dst_en.
171 * true(1): enable false(0): disable
172 * if you don't want to use linked list mode, please leave llpx = 0.
173 */
174 void *llpx;
175 bool llp_src_en;
176 bool llp_dst_en;
177
178 /*
179 * these parameters are used to config the gather feature of SNPS AHB
180 * DMAC. if you don't what you are doing, please leave these arguments
181 * with 0.
182 * src_gth_cnt: Source contiguous transfer count between
183 * successive gather boundaries.
184 * src_gth_intvl: interval of contiguous source gather block.
185 */
186 uint32_t src_gth_cnt;
187 uint32_t src_gth_intvl;
188
189 /*
190 * these parameters are used to config the scatter feature of SNPS AHB
191 * DMAC. if you don't what you are doing, please leave these arguments
192 * with 0.
193 * dst_sct_cnt: Destination contiguous transfer count between
194 * successive scatter boundaries.
195 * dst_sct_intvl: interval of contiguous destination scatter block.
196 */
197 uint32_t dst_sct_cnt;
198 uint32_t dst_sct_intvl;
199
200 /*
201 * Whether enable auto-reload multi-blocks transfer.
202 * 0 = disable 1 = enable
203 */
204 bool src_auto_reload;
205 bool dst_auto_reload;
206
207 /*
208 * Driver will allocate automatically LLI array to finish this dma
209 * xfer depending on the length of this transfer
210 */
211 void *lli;
212
213 /*
214 * Record the number of Linked list items of current transfer.
215 */
216 uint32_t lli_num;
217} DmacChannelCfg;
